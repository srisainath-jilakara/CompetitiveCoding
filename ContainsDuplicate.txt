class Solution {
    public boolean containsDuplicate(int[] nums) {
        Arrays.sort(nums);
        for(int i=0;i<nums.length;i++)
        {
            if(i+1<nums.length)
            {
                if(nums[i]==nums[i+1])
                    return true;
                else
                    continue;
            }
        }
        return false;
    }
}
